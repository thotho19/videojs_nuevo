Code Setup Source (Watermark logo)

<!--Head tag CSS-->
<link href="../videojs/css/videojs.min.css" rel="stylesheet" type="text/css" />

<!--Loading videojs and nuevo plugin-->
<script type="text/javascript" src="../videojs/video.min.js"></script>
<script type="text/javascript" src="../videojs/nuevo.min.js"></script>

<!--Video tag setup-->
<video id="player_one" class="video-js vjs-fluid vjs-default-skin" controls preload playsinline poster="images/poster.jpg">
	<source src="//devnuevo.com/media/video/demo_720.mp4" type="video/mp4">
	<source src="//devnuevo.com/media/video/demo_720.webm" type="video/webm">
</video>

<!--On page player and plugin initialization-->
<script>
var player = videojs('player_1');
player.nuevoPlugin({ 
	logotitle:'Nuevo plugin for Videojs',
	logo: 'images/logo.png',
	logoposition:'RT',
	logourl:'https://www.nuevodevel.com/nuevo/',
});
</script>
